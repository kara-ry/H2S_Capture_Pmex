#SBATCH --partition=kamiak
#SBATCH --job-name=duplmarkt
#SBATCH --mem=40G

M=/data/kelley/projects/pmex/capture/nick_alignmentTwo/nick_dplmarkmetrics/metric.${theNumber}.txt
